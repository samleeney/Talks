% Slide 2: Method
\begin{frame}{cNBRE: a classifier that outputs $\log K(\mathbf d\mid \alpha)$}
\small
\begin{columns}[T]
\begin{column}{0.42\textwidth}
\begin{itemize}\itemsep2pt
\item Simulate training data under $M_1$ and $M_0$ across design space $\alpha$.
\item Train $f_\phi(\mathbf d,\alpha)$ with exponential loss
\[
\ell = \mathbb E_{M_1}\!\left[e^{-f_\phi/2}\right]+\mathbb E_{M_0}\!\left[e^{+f_\phi/2}\right].
\]
\item Optimum: $f_\phi^*(\mathbf d,\alpha)=\log K(\mathbf d\mid \alpha)$ (with balanced classes).
\item Cost: train once (4--10 A100 GPU-hours), then evaluate in ms for any $\alpha$.
\end{itemize}
\end{column}
\begin{column}{0.55\textwidth}
\centering

% Slide 4: REACH setup + tau plot
\begin{frame}{REACH: detection vs observation time}
\small
\begin{columns}[T]
\begin{column}{0.47\textwidth}
\textbf{Forward models}
\begin{align*}
M_1:\ \mathbf d &= \mathbf d_{\mathrm{fg}} + \mathbf d_{21} + \mathbf n\\
M_0:\ \mathbf d &= \mathbf d_{\mathrm{fg}} + \mathbf n
\end{align*}

\textbf{Design parameters} $\alpha=(\tau,\psi)$.

\vspace{2pt}
\textbf{Detection probability}
\[
P_{\mathrm{det}}(\alpha) = \Pr_{M_1,\alpha}\!\big[\log K(\mathbf d\mid\alpha)>\log K_{\mathrm{crit}}\big].
\]

\begin{itemize}\itemsep1pt
\item $\log K_{\mathrm{crit}}$: $\sigma$-equivalent FPR under $M_0$.
\item $P_{\mathrm{det}}$ plateaus after ${\sim}10\,\mathrm{min}$ due to a \textbf{systematics noise floor}.
\end{itemize}
\end{column}

% Slide: A note on cost
\begin{frame}{A note on cost}
\small
\begin{columns}[T]
\begin{column}{0.55\textwidth}
\begin{itemize}\itemsep4pt
\item GPUs are not free --- but physical experiments cost \textit{far} more for comparable sensitivity gains.
\item cNBRE training: \textbf{4--10 A100 GPU-hours} (one-time).
\item Physical alternatives for similar $\Delta P_{\mathrm{det}}$:
\begin{itemize}\itemsep1pt
    \item New antenna design \& fabrication
    \item Lower-loss cables \& connectors
    \item Additional calibration hardware
    \item Extended observation campaigns
\end{itemize}
\item SKA: ${\sim}$\texteuro2\,B. LHC: ${\sim}$\$9\,B.
\item $\sim$20pp gain from antenna orientation costs \textbf{nothing} once identified.
\end{itemize}
\end{column}
