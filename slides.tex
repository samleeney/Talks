\section{Method}

\begin{frame}{\small{What quantity do we need for detection?}}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.56\textwidth}
\small
We quantify confidence in a detection by comparing the Bayesian evidence for a signal model versus a null model:
\begin{align*}
M_1 &: \text{signal + foreground + noise} \\
M_0 &: \text{foreground + noise}
\end{align*}

\vspace{-0.16cm}
\[
\red{K(\mathbf d)}
=
\frac{p(\mathbf d \mid M_1)}{p(\mathbf d \mid M_0)}.
\]

\vspace{-0.12cm}
\begin{itemize}
\item $K > 1$: data prefer the signal model.
\item $K \gg 1$: strong evidence for the signal.
\item If I can compute $K$, I can tell which design gives the best chances of detection.
\end{itemize}
\end{column}
\begin{column}{0.40\textwidth}
\begin{tcolorbox}[keybox,title=The bottleneck]
\footnotesize
Optimising a design parameter requires many samples over the full design space.

\vspace{0.08cm}
Each sample is an independent sampling run.

\vspace{0.08cm}
100 design choices $\times$ 100 possible signals $=$ \textbf{10,000} expensive runs.

\vspace{0.08cm}
Add one more parameter and it becomes \textbf{1,000,000}.
\end{tcolorbox}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{\small{How do evidence networks avoid that bottleneck?}}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.50\textwidth}
\footnotesize
If we have a simulator for $M_1$ and $M_0$, we can train a classifier $f_\phi(\mathbf d)$ to distinguish them.

\vspace{0.04cm}
With an appropriately chosen loss function, the network converges on the Bayes factor:
\[
\mathcal{L}
=
\mathbb E_{M_1}\!\left[e^{-f_\phi(\mathbf d)/2}\right]
+
\mathbb E_{M_0}\!\left[e^{+f_\phi(\mathbf d)/2}\right].
\]

\vspace{-0.10cm}
\[
f_\phi^\star(\mathbf d)=\log K(\mathbf d).
\]

\vspace{-0.05cm}
\begin{itemize}
\item Feed the network different data, it outputs the Bayes factor.
\item Train once, evaluate in milliseconds.
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\centering
\resizebox{\linewidth}{!}{%
\begin{tikzpicture}[
    node distance=0.6cm and 1.0cm,
    every node/.style={font=\footnotesize},
    input/.style={circle, draw, minimum size=0.85cm, thick},

\end{column}
\end{columns}

\begin{frame}{\small{How does this become forecasting?}}
\begin{tcolorbox}[resultbox]
\small
\textbf{Given some range of physics parameters, what is our probability of detecting the signal?}
\end{tcolorbox}

\vspace{0.02cm}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.52\textwidth}
\footnotesize
Gessey-Jones et al.'s key insight was to use the Bayes factor as a forecasting utility:

\vspace{0.04cm}
\[
\pdet
=
\Pr\!\bigl[\log K(\mathbf d) > \log K_{\mathrm{crit}} \,\big|\, M_1\bigr].
\]

\vspace{-0.08cm}
\begin{itemize}
\item Simulate many realisations under $M_1$.
\item Evaluate $\log K$ for each one.
\item Count the fraction that cross the threshold.
\end{itemize}
\end{column}
\begin{column}{0.44\textwidth}
\centering
\begin{tikzpicture}[
  node distance=0.58cm,
  every node/.style={font=\footnotesize, align=center},
  box/.style={draw, rounded corners=3pt, minimum width=2.5cm, minimum height=0.72cm, thick},
  arr/.style={-{Stealth[length=2.2mm]}, thick}
]
\node[box, fill=green!15] (sims) {$M_1$ simulations};
\node[box, fill=blue!10, below=of sims] (eval) {evaluate\\$\log K(\mathbf d)$};
\node[box, fill=red!15, below=of eval] (frac) {fraction above\\$\log K_{\mathrm{crit}}$};
\draw[arr] (sims) -- (eval);
\draw[arr] (eval) -- (frac);
\node[font=\scriptsize\bfseries, red!70!black, below=0.18cm of frac] {$=\ \pdet$};
\end{tikzpicture}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{\small{What changes in conditional neural Bayes ratio estimation?}}
\footnotesize
\begin{columns}[T,onlytextwidth]
\begin{column}{0.41\textwidth}
Extend the classifier: predict which model the data came from \textbf{given some design parameter}:
\[
f_\phi(\mathbf d,\alpha)
\approx
\log K(\mathbf d \mid \alpha).
\]

\vspace{-0.18cm}
\begin{itemize}\itemsep1pt
\item e.g.\ ``did this data come from the signal model, \textbf{given some cable length}?''
\item Train once across the full design space.
\item Query \textbf{any} $\alpha$ in milliseconds.
\end{itemize}

\vspace{0.02cm}
\begin{tcolorbox}[resultbox,title=Key idea]
\footnotesize
\textbf{Amortised:} check how ANY design choice impacts detection odds.
\end{tcolorbox}
\end{column}

\section{Applications}

\begin{frame}{\small{What do you need to use cNBRE?}}
\vfill
\begin{center}
\begin{tcolorbox}[keybox, width=0.80\textwidth]
\normalsize
\centering
All you need is a \textbf{forward simulator} and a \textbf{design parameter} you want to optimise.
\end{tcolorbox}
\end{center}

\vspace{0.3cm}
\small
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
Most experiments that do parameter estimation \textbf{already have} a forward model inside their likelihood.

\vspace{0.1cm}
\begin{enumerate}
\item Define your two models ($M_1$ vs $M_0$).
\item Identify your design parameters $\alpha$.
\item Simulate training data across the design space.
\item Train a single cNBRE network.
\end{enumerate}
\end{column}
\begin{column}{0.48\textwidth}
\begin{tcolorbox}[resultbox]
\small
The simulator does not need to be differentiable, fast, or GPU-based.

\vspace{0.05cm}
It just needs to produce realistic mock data for any $\alpha$.
\end{tcolorbox}
\end{column}
\end{columns}
\vfill
\end{frame}

\begin{frame}{\small{Derivation: optimal network output is the log-Bayes factor}}
\footnotesize
Minimise the expected exponential loss:
\[
\mathcal{J}[f] = \int \left[ p(\mathbf{d}|M_1)\, e^{-f(\mathbf{d})/2} + p(\mathbf{d}|M_0)\, e^{f(\mathbf{d})/2} \right] \mathrm{d}\mathbf{d}.
\]
Set the functional derivative to zero:
\[
\frac{\delta \mathcal{J}}{\delta f} = -\tfrac{1}{2}\, p(\mathbf{d}|M_1)\, e^{-f^*/2} + \tfrac{1}{2}\, p(\mathbf{d}|M_0)\, e^{f^*/2} = 0.
\]
Rearranging:
\[
e^{f^*(\mathbf{d})} = \frac{p(\mathbf{d}|M_1)}{p(\mathbf{d}|M_0)} = K(\mathbf{d})
\qquad \Longrightarrow \qquad
\boxed{f^*(\mathbf{d}) = \log K(\mathbf{d})}.
\]
\end{frame}
