\begin{frame}{\small{Step 3: threshold on compressed S/N}}
    \begin{itemize}
        \item At each trial DM, the 2D dynamic spectrum is collapsed to a 1D S/N time series.
        \item A candidate is flagged if S/N exceeds a fixed threshold.
        \item The full time-frequency structure, the thing that distinguishes a real pulse from RFI, is gone.
    \end{itemize}
    \vspace{0.3cm}
    \begin{tcolorbox}[colback=red!5!white,colframe=red!75!black,title=The problem]
        Anomaly detection and signal searching are completely decoupled, and the data is compressed before you ever ask ``is there a signal here?''
    \end{tcolorbox}
\end{frame}

\begin{frame}{\small{What does the standard pipeline throw away?}}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Standard approach}
            \begin{enumerate}
                \item Flag RFI (independent step).
                \item Dedisperse $\rightarrow$ compress to 1D S/N time series.
                \item Threshold on S/N.
            \end{enumerate}
            \vspace{0.2cm}
            Anomaly detection and searching are \textbf{decoupled}; the 2D time-frequency structure is discarded before you ever ask ``is there a signal?''
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{What we propose}
            \begin{enumerate}
                \item Work directly on the raw 2D dynamic spectrum.
                \item Fit signal and flag anomalies \textbf{jointly} in one likelihood.
                \item Detection via Bayesian model comparison.
            \end{enumerate}
            \vspace{0.2cm}
            No compression, no information loss, and RFI handling is \textbf{fully automated} inside the model rather than a separate manual pre-processing step.
        \end{column}
    \end{columns}
\end{frame}

\section{Why Bayesian?}
\begin{frame}{\small{Why is the evidence useful?}}
    \begin{itemize}
        \item The evidence lets us compare models, not just fit parameters inside one model.
        \item A model with too many free parameters is penalised automatically (Occam's razor).
        \item Formally, the Bayes factor between two models is
        \[
        \mathcal{B} = \frac{\mathcal{Z}_1}{\mathcal{Z}_0}.
        \]
        \item $\mathcal{B}>1$ favours model 1; $\mathcal{B}<1$ favours model 0.
    \end{itemize}
\end{frame}

\begin{frame}{\small{Bayes factor for detection}}
    \begin{tcolorbox}[colback=blue!5!white,colframe=blue!75!black,title=Detection criterion]
        Compare two models: $M_1$ (signal + noise) vs $M_0$ (noise only).
    \end{tcolorbox}
    \vspace{0.3cm}
    \[
    \mathcal{B} = \frac{\mathcal{Z}_{\text{signal+noise}}}{\mathcal{Z}_{\text{noise-only}}} \gtrless 1
    \]
    \vspace{0.2cm}
    \begin{itemize}
        \item $\mathcal{B}=1$ is the break-even point between models.
        \item In practice we use $\mathcal{B}>\mathcal{B}_{\mathrm{thr}}$, with $\mathcal{B}_{\mathrm{thr}}$ calibrated by injection-recovery at a target false-alarm rate.
        \item The key practical question: can we compute these evidences fast enough for survey data rates?
    \end{itemize}
\end{frame}

\begin{frame}{\small{What can we encode as prior knowledge?}}
    \begin{itemize}
        \item We know things about transients before looking at a given chunk of data.
        \item Encode this as priors on the search parameters:
        \begin{itemize}
            \item DM range from expected source population or sky direction.
            \item Pulse width from instrumental resolution and physical limits.
            \item Arrival time: uniform across the observation window.
        \end{itemize}
        \item The prior concentrates the search where signals are physically plausible.
        \item This is the key advantage over matched filtering: the prior does useful work.
    \end{itemize}
\end{frame}

\section{Bayesian transient model}
\begin{frame}{\small{Key message}}
    \begin{tcolorbox}[colback=blue!5!white,colframe=blue!75!black,title=Key message]
        Search on the raw time-frequency cube, not on a compressed S/N time series. Fit the signal and flag anomalies jointly in one model.
    \end{tcolorbox}
    \vspace{0.35cm}
    \begin{itemize}
        \item \textbf{Prior} $\rightarrow$ DM range, pulse width, arrival window from physics.
        \item \textbf{Likelihood} $\rightarrow$ generative model on the full 2D dynamic spectrum (no dedisperse-then-threshold).
        \item \textbf{Evidence} $\mathcal{Z}$ $\rightarrow$ Bayes factor flags ``signal present'' vs ``noise only''.
        \item The math parallelises; GPU acceleration makes this practical.
    \end{itemize}
\end{frame}

\begin{frame}{\small{Generative model}}
    \[
    I_{ij}
    =
    \frac{A}{\sqrt{2\pi}\,\sigma_{\mathrm{frb}}'}
    \exp\!\left(
      -\frac{\big[t_j - t_0' - K\,\mathrm{DM}'/f_i^2\big]^2}{2\,{\sigma_{\mathrm{frb}}'}^2}
    \right)
    + \epsilon_{ij},
    \]
    \begin{itemize}
        \item Likelihood: $\epsilon_{ij} \sim \mathcal{N}(0, \sigma_{\text{noise}}^2)$ evaluated on the full cube.
        \item Parameters: amplitude $A$, arrival time $t_0'$, dispersion measure $\mathrm{DM}'$, pulse width $\sigma_{\mathrm{frb}}'$.
        \item Inference target: posteriors on all parameters, plus evidence $\mathcal{Z}$ for detection.
    \end{itemize}
\end{frame}

\section{Bayesian anomaly detection}
\begin{frame}{\small{Anomaly model: Bernoulli mask}}
    \begin{itemize}
        \item We don't know which pixels are RFI, so we let the model decide. Each pixel gets a hidden label: nominal or anomalous.
        \item If nominal, the pixel is evaluated under the signal model. If anomalous, we say ``this pixel could be anything'' (flat likelihood over a broad scale $\Delta$).
        \item Na\"ively this means summing over $2^{N_{\mathrm{pix}}}$ possible masks, but because pixels are independent the sum factorises into one term per pixel:
        \[
        \log P(\mathbf{I}\mid\theta,p)=\sum_{ij}\log\!\left((1-p)L_{ij}(\theta)+\frac{p}{\Delta}\right).
        \]
        \item One likelihood that simultaneously fits the pulse and identifies RFI. The posterior on $p$ tells you how much of the data was contaminated.
    \end{itemize}
\end{frame}

\begin{frame}{\small{Flagging detections}}
    \begin{itemize}
        \item Operationally we flag when the Bayes factor exceeds a calibrated threshold:
        \[
        \mathcal{B} = \frac{\mathcal{Z}_{\text{signal+anomaly}}}{\mathcal{Z}_{\text{noise-only}}} > \mathcal{B}_{\mathrm{thr}}.
        \]
        \item $\mathcal{B}_{\mathrm{thr}}$ is set by injection-recovery for a chosen false-alarm rate.
        \item Posterior on $p$ quantifies how much of the cube was treated as interference.
    \end{itemize}
\end{frame}

\begin{frame}{\small{Problem\ldots\ too slow}}
    \begin{itemize}
        \item Current full nested-sampling inference is typically $\sim$1-3 minutes per candidate window.
        \item A real-time trigger pipeline needs $\lesssim$50 ms per candidate decision.
        \item We are therefore roughly $10^3$ too slow without analytic marginalisation.
    \end{itemize}
\end{frame}

\section{Fast analytic evidence}
\begin{frame}{\small{Goal: closed-form Bayes factor}}
    \[
    \mathcal{B} \equiv \frac{\mathcal{Z}_1}{\mathcal{Z}_0}, \qquad
    \mathcal{Z}_m = \int p(I\mid\theta, M_m)\,p(\theta\mid M_m)\,d\theta
    \]
    \begin{itemize}
        \item Treat template parameters as nuisance; integrate them out analytically where possible.
        \item Keep only a small numerical search over $(\mathrm{DM}', \sigma_{\mathrm{frb}}')$ and discrete $t_0'$.
        \item Result: fast, calibrated evidence for detection.
    \end{itemize}
\end{frame}

\begin{frame}{\small{Signal model}}
    \[
    I_{ij}
    =
    \frac{A}{\sqrt{2\pi}\,\textcolor{red}{\sigma_{\mathrm{frb}}'}}
    \exp\!\left(
      -\frac{\big[t_j - \textcolor{red}{t_0'} - K\,\textcolor{red}{\mathrm{DM}'}/f_i^2\big]^2}{2\,\textcolor{red}{{\sigma_{\mathrm{frb}}'}^2}}
    \right)
    + \epsilon_{ij}, \quad \epsilon_{ij}\sim\mathcal{N}(0,\sigma^2)
    \]
    \begin{itemize}
        \item Parameters to marginalise: $t_0'$ (discrete), amplitude $A$, noise $\sigma$, track shape $(\mathrm{DM}',\sigma_{\mathrm{frb}}')$.
    \end{itemize}
\end{frame}

\begin{frame}{\small{Visualising $t_0'$ (arrival asymptote)}}
    \centering
    \includegraphics[width=0.9\textwidth,height=0.78\textheight,keepaspectratio]{images/t0_visual.png}
\end{frame}

\begin{frame}{\small{Marginalising $t_0'$ (discrete time index)}}
    \[
    I_{ij}
    =
    \frac{A}{\sqrt{2\pi}\,\sigma_{\mathrm{frb}}'}
    \exp\!\left(
      -\frac{\big[t_j - \textcolor{red}{t_0'} - K\,\mathrm{DM}'/f_i^2\big]^2}{2\,{\sigma_{\mathrm{frb}}'}^2}
    \right)
    + \epsilon_{ij}, \quad \epsilon_{ij}\sim\mathcal{N}(0,\sigma^2)
    \]
    \begin{itemize}
        \item Treat $t_0'$ as a discrete prior over time bins (often uniform).
        \item Condition on $t_0'$ and evaluate in parallel across time bins:
        \[
        \mathcal{Z}_1(t_0') = \int p(I\mid t_0',A,\sigma,\phi)\,p(A,\sigma,\phi)\,dA\,d\sigma\,d\phi
        \]
        with $\phi=(\mathrm{DM}',\sigma_{\mathrm{frb}}')$.
        \item Full evidence: $\mathcal{Z}_1 = \sum_{t_0'} p(t_0')\,\mathcal{Z}_1(t_0')$; embarrassingly parallel over $t_0'$.
    \end{itemize}
\end{frame}

\begin{frame}{\small{Marginalising amplitude $A$}}
    \[
    I_{ij}
    =
    \frac{\textcolor{red}{A}}{\sqrt{2\pi}\,\sigma_{\mathrm{frb}}'}
    \exp\!\left(
      -\frac{\big[t_j - t_0' - K\,\mathrm{DM}'/f_i^2\big]^2}{2\,{\sigma_{\mathrm{frb}}'}^2}
    \right)
    + \epsilon_{ij}, \quad \epsilon_{ij}\sim\mathcal{N}(0,\sigma^2)
    \]
    \begin{itemize}
        \item The amplitude $A$ just scales the template, so it appears linearly in the model.
        \item This means we can choose a Gaussian prior on $A$ and integrate it out exactly. No sampling needed.
        \item After integrating, the likelihood no longer depends on $A$ at all. One parameter gone, no approximation.
    \end{itemize}
\end{frame}

\begin{frame}{\small{Marginalising noise $\sigma^2$}}
    \[
    I_{ij}
    =
    \frac{A}{\sqrt{2\pi}\,\sigma_{\mathrm{frb}}'}
    \exp\!\left(
      -\frac{\big[t_j - t_0' - K\,\mathrm{DM}'/f_i^2\big]^2}{2\,{\sigma_{\mathrm{frb}}'}^2}
    \right)
    + \epsilon_{ij}, \quad \epsilon_{ij}\sim\mathcal{N}(0,\textcolor{red}{\sigma^2})
    \]
    \begin{itemize}
        \item Same idea again: choose an inverse-gamma prior on $\sigma^2$ so the integral has a closed-form solution.
        \item After integrating out both $A$ and $\sigma^2$, we are left with:
        \[
        p(I\mid t_0',\phi) \propto \big(\beta_0 + \tfrac{1}{2}S\big)^{-(\alpha_0 + N_{\mathrm{pix}}/2)}.
        \]
        \item Two nuisance parameters gone, no sampler needed. All that remains is the track shape: DM and pulse width.
    \end{itemize}
\end{frame}

\begin{frame}{\small{Laplace for $(\mathrm{DM}',\sigma_{\mathrm{frb}}')$}}
    \[
    I_{ij}
    =
    \frac{A}{\sqrt{2\pi}\,\textcolor{red}{\sigma_{\mathrm{frb}}'}}
    \exp\!\left(
      -\frac{\big[t_j - t_0' - K\,\textcolor{red}{\mathrm{DM}'}/f_i^2\big]^2}{2\,{\textcolor{red}{\sigma_{\mathrm{frb}}'}}^2}
    \right)
    + \epsilon_{ij}, \quad \epsilon_{ij}\sim\mathcal{N}(0,\sigma^2)
    \]
    \begin{itemize}
        \item Remaining continuous track parameters $\phi = (\mathrm{DM}', \sigma_{\mathrm{frb}}')$.
        \item Laplace approximation (local Gaussian near the best-fit point):
        \[
        \log \mathcal{Z}_1(t_0') \approx \ell(\hat{\phi}) + \frac{d}{2}\log(2\pi) - \tfrac{1}{2}\log|H|,\quad d=2.
        \]
        \item Sum over $t_0'$ and form $\mathcal{B} = \mathcal{Z}_1/\mathcal{Z}_0$ for a fast Bayes factor per candidate.
    \end{itemize}
\end{frame}

\begin{frame}{\small{Summary}}
    \begin{tcolorbox}[colback=blue!5!white,colframe=blue!75!black,title=Summary]
        A fully Bayesian transient search on raw dynamic spectra, with analytic marginalisation fast enough for real-time use.
    \end{tcolorbox}
    \vspace{0.35cm}
    \begin{itemize}
        \item Physics-informed priors concentrate the search where signals are plausible.
        \item Joint signal + anomaly model handles RFI fully automatically, with no separate flagging step.
        \item Analytic marginalisation over nuisance parameters gives a fast Bayes factor.
        \item Thresholds are calibrated via injections, not fixed at the break-even $\mathcal{B}=1$ point.
    \end{itemize}
\end{frame}
