%% Bayesian approach: marginalise, but 2^N is intractable %%
\begin{frame}{\small{How do we find the right mask?}}
    \footnotesize
    We take a Bayesian approach: marginalise over \textbf{all possible masks} to get a probability per point that it is contaminated.

    \vspace{0.2cm}

    Write the likelihood blending normal and anomaly terms:
    \[
    P(\mathcal{D}, \vec{\varepsilon}\mid\theta) = \prod_{i} \left[(1-p_i)\,\mathcal{L}_i(\theta)\right]^{1-\varepsilon_i} \left[\frac{p_i}{\Delta}\right]^{\varepsilon_i}
    \]

    In principle we sum over all $2^N$ mask configurations. This quickly becomes intractable.

    \vspace{0.2cm}

    \begin{tcolorbox}[colback=blue!5!white,colframe=blue!75!black,title=Dominant mask approximation]
        \scriptsize
        Instead of summing, we \textbf{maximise}: at each step of the sampler, pick the best mask for the current $\theta$. Because the pixels factorise, this reduces to a per-point comparison:
        \[
        \varepsilon_i^\star = 1 \quad \text{if} \quad \frac{p_i}{\Delta} > (1-p_i)\,\mathcal{L}_i(\theta)
        \]
    \end{tcolorbox}

    \centering
    What does this look like in practice?
\end{frame}

            {\scriptsize The resulting log-likelihood with the dominant mask:}
            {\small
            \[
            \log P = \sum_i \max\!\left\{\log\mathcal{L}_i + \log(1{-}p_i),\; \log p_i - \log\Delta\right\}
            \]
            }

            \vspace{-0.2cm}
            {\tiny \textbf{Note:} $\log\Delta$ acts as an \textbf{Occam penalty}. Flagging a point costs $\log\Delta$; the model only flags when the likelihood gain outweighs this cost. The threshold $p_i$ is \textbf{fit as a model parameter}, not set by hand.}

%% Logit falls out of the likelihood %%
\begin{frame}{\small{A familiar activation function}}
    \scriptsize
    Take logs of the per-point decision rule from our likelihood:
    \[
    \frac{p_i}{\Delta} > (1-p_i)\,\mathcal{L}_i
    \qquad\Longrightarrow\qquad
    \log p_i - \log(1-p_i) > \log\mathcal{L}_i + \log\Delta
    \]
    \[
    \boxed{\;\mathrm{logit}(p_i) > \log\mathcal{L}_i + \log\Delta\;}
    \]

    Invert with the sigmoid $\sigma(x) = 1/(1+e^{-x})$ to get the posterior anomaly probability:
    \[
    P_{\mathrm{anom},i} = \sigma\!\big(\mathrm{logit}(p_i) - \log\mathcal{L}_i - \log\Delta\big)
    \]

    \begin{tcolorbox}[colback=blue!5!white,colframe=blue!75!black]
        \scriptsize
        The logit and sigmoid fall directly out of our likelihood. These are standard activation functions in machine learning. The same framework that works inside nested sampling can therefore be made differentiable and plugged into a neural network.
    \end{tcolorbox}
\end{frame}

\begin{frame}{\small{Hard vs.\ soft mixture models}}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.57\textwidth}
            {\scriptsize
            \textbf{Hard mixture: Bayesian inference}
            \[
            \log p(\mathcal{D}\mid\theta) = \sum_i \max\left\{\log \mathcal{L}_i + \log(1-p_i),\; \log p_i - \log \Delta_i\right\}
            \]

            \vspace{0.15cm}

            \textbf{Soft mixture: machine learning}
            \[
            \log p(\mathcal{D}\mid\theta) = \sum_i \log\left[(1-p_i)\mathcal{L}_i + \frac{p_i}{\Delta_i}\right]
            \]
            }

            \begin{tcolorbox}[colback=blue!5!white,colframe=blue!75!black]
                \scriptsize
                The hard $\max$ becomes a differentiable log-sum-exp. The corner between the flat floor and the likelihood has no gradient; the soft version smooths this, letting the same idea move into gradient-based machine learning.
            \end{tcolorbox}
        \end{column}
        \begin{column}{0.39\textwidth}
            \centering
            \includegraphics[width=0.80\textwidth]{images/hard_mixture_model.png}

            \vspace{0.18cm}

            \includegraphics[width=0.80\textwidth]{images/soft_mixture_model.png}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{\small{Full derivation: Bayesian anomaly detection}}
    \scriptsize
    \textbf{1.} Likelihood per data point under the model: $\mathcal{L}_i(\theta)$.

    \vspace{0.1cm}
    \textbf{2.} Introduce a Bernoulli mask $\varepsilon_i \in \{0,1\}$ with prior anomaly rate $p_i$ and anomaly scale $\Delta$:
    \vspace{-0.15cm}
    \[
    P(\mathcal{D}, \vec{\varepsilon}\mid\theta) = \prod_{i} \left[(1-p_i)\,\mathcal{L}_i(\theta)\right]^{1-\varepsilon_i} \left[\frac{p_i}{\Delta}\right]^{\varepsilon_i}
    \]
    \vspace{-0.25cm}

    \textbf{3.} Marginalise the mask. Pixels factorise, so the $2^N$ sum is tractable in principle:
    \vspace{-0.15cm}
    \[
    P(\mathcal{D}\mid\theta) = \prod_{i} \left[(1-p_i)\,\mathcal{L}_i + \frac{p_i}{\Delta}\right]
    \]
    \vspace{-0.25cm}

    \textbf{4.} Dominant mask approximation (maximise instead of sum):
    \vspace{-0.15cm}
    \[
    \log P \approx \sum_i \max\!\left\{\log\mathcal{L}_i + \log(1{-}p_i),\;\; \log p_i - \log\Delta\right\}
    \]
    \vspace{-0.25cm}

    \textbf{5.} Posterior anomaly probability per point (Bayes' rule on mask states):
    \vspace{-0.15cm}
    \[
    P_{\mathrm{anom},i} = \frac{p_i/\Delta}{(1-p_i)\,\mathcal{L}_i + p_i/\Delta}
    \]
\end{frame}

\begin{frame}{\small{Full derivation: logit and sigmoid}}
    \scriptsize
    \textbf{6.} Take logs of the per-point decision rule:
    \vspace{-0.1cm}
    \[
    \frac{p_i}{\Delta} > (1-p_i)\,\mathcal{L}_i
    \quad\Longrightarrow\quad
    \log\frac{p_i}{1-p_i} > \log\mathcal{L}_i + \log\Delta
    \quad\Longrightarrow\quad
    \mathrm{logit}(p_i) > \log\mathcal{L}_i + \log\Delta
    \]
    \vspace{-0.1cm}

    \textbf{7.} Rearrange $P_{\mathrm{anom},i}$ by dividing top and bottom by $p_i/\Delta$:
    \vspace{-0.1cm}
    \[
    P_{\mathrm{anom},i} = \frac{1}{1 + \frac{(1-p_i)\,\mathcal{L}_i\,\Delta}{p_i}}
    = \frac{1}{1 + \exp\!\big(\log\mathcal{L}_i + \log\Delta - \mathrm{logit}(p_i)\big)}
    = \sigma\!\big(\mathrm{logit}(p_i) - \log\mathcal{L}_i - \log\Delta\big)
    \]
    \vspace{-0.1cm}

    where $\sigma(x) = 1/(1 + e^{-x})$ is the sigmoid function.

    \vspace{0.2cm}

    \textbf{8.} Soft mixture (ML extension): replace $\max$ with $\mathrm{logsumexp}$:
    \vspace{-0.1cm}
    \[
    \log P(\mathcal{D}\mid\theta) = \sum_i \log\left[(1-p_i)\,\mathcal{L}_i + \frac{p_i}{\Delta}\right]
    \]
